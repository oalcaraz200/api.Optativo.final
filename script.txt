-- Database: postgres7

-- DROP DATABASE IF EXISTS postgres7;

CREATE DATABASE postgres7
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Spanish_Paraguay.1252'
    LC_CTYPE = 'Spanish_Paraguay.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;



------------------------------------


-- Table: public.cuenta

-- DROP TABLE IF EXISTS public.cuenta;

CREATE TABLE IF NOT EXISTS public.cuenta
(
    id bigint NOT NULL,
    nombre_cuenta character varying(60) COLLATE pg_catalog."default",
    saldo double precision,
    saldo_limite double precision,
    limite_transferencia double precision,
    estado character varying(20) COLLATE pg_catalog."default",
    id_cuenta bigint NOT NULL,
    numero_cuenta character varying(60) COLLATE pg_catalog."default",
    moneda character varying(40) COLLATE pg_catalog."default",
    CONSTRAINT cuenta_pkey PRIMARY KEY (id_cuenta)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.cuenta
    OWNER to postgres;

--------------------------------------------------

-- Table: public.extracto

-- DROP TABLE IF EXISTS public.extracto;

CREATE TABLE IF NOT EXISTS public.extracto
(
    id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
    operacion character varying(60) COLLATE pg_catalog."default",
    observacion character varying(200) COLLATE pg_catalog."default",
    importe double precision,
    estado character varying(60) COLLATE pg_catalog."default",
    nro_cuenta character varying(60) COLLATE pg_catalog."default",
    fecha character varying(60) COLLATE pg_catalog."default",
    CONSTRAINT extracto_pkey PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.extracto
    OWNER to postgres;


---------------------------------------------------


-- Table: public.persona

-- DROP TABLE IF EXISTS public.persona;

CREATE TABLE IF NOT EXISTS public.persona
(
    id bigint NOT NULL,
    nombre character varying(60) COLLATE pg_catalog."default",
    apellido character varying(60) COLLATE pg_catalog."default",
    edad bigint,
    email character varying(60) COLLATE pg_catalog."default",
    telefono character varying(60) COLLATE pg_catalog."default",
    CONSTRAINT "Persona_pkey" PRIMARY KEY (id)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.persona
    OWNER to postgres;


--------------------------------------------------



-- Table: public.usuarios

-- DROP TABLE IF EXISTS public.usuarios;

CREATE TABLE IF NOT EXISTS public.usuarios
(
    id bigint NOT NULL,
    id_persona bigint NOT NULL,
    usuario character varying(60) COLLATE pg_catalog."default",
    password character varying(60) COLLATE pg_catalog."default",
    estado character varying(30) COLLATE pg_catalog."default",
    CONSTRAINT usuarios_pkey PRIMARY KEY (id_persona)
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.usuarios
    OWNER to postgres;
